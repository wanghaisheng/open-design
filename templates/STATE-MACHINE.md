## 实现示例

```javascript
// 状态机实现示例
const stateMachine = {
  currentState: 'state1',
  states: {
    state1: {
      on: {
        event1: 'state2'
      },
      onEnter: () => { /* 进入动作 */ },
      onExit: () => { /* 退出动作 */ }
    },
    state2: {
      on: {
        event2: 'state3'
      },
      onEnter: () => { /* 进入动作 */ },
      onExit: () => { /* 退出动作 */ }
    },
    state3: {
      on: {
        event3: 'state1'
      },
      onEnter: () => { /* 进入动作 */ },
      onExit: () => { /* 退出动作 */ }
    }
  },
  transition(event) {
    const currentState = this.states[this.currentState];
    const nextState = currentState.on[event];
    if (nextState) {
      currentState.onExit();
      this.currentState = nextState;
      this.states[nextState].onEnter();
    }
  }
};
```
